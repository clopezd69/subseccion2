         y ="Muertes", 
       title= "Número de muertes por aerolinea 1985-2014")+
  theme(legend.position = "bottom")+
  guides(fill = guide_legend(title = "Periodos"))+
  scale_fill_hue(labels = c("2000-2014", "1985-1999"))
